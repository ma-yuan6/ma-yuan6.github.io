import numpy as np

# ÂÆö‰πâ Jacobi Ëø≠‰ª£ÂáΩÊï∞
def jacobi_iteration(A, b, N=10, x=None):
    if x is None:
        x = np.zeros_like(b)
    D = np.diag(np.diag(A))
    R = A - D
    for i in range(N):
        x = np.dot(np.linalg.inv(D), b - np.dot(R, x))
        print("Ëø≠‰ª£ ", i+1, ": ", x)
    return x

# ËÆæÁΩÆÁ≥ªÊï∞Áü©Èòµ A ÂíåÂ∏∏Êï∞ÂêëÈáè b

import numpy as np

# ÂÆö‰πâ Gauss-Seidel Ëø≠‰ª£ÂáΩÊï∞
def gauss_seidel(A, b, eps=1e-10, max_iter=1000):
    n = len(b)
    x = np.zeros(n)
    for num_iter in range(max_iter):
        for i in range(n):
            s = np.dot(A[i][:i], x[:i])
            t = np.dot(A[i][i+1:], x[i+1:])
            x[i] = (b[i] - s - t) / A[i][i]
        print("Ëø≠‰ª£", num_iter+1, ":", x)
        if np.allclose(np.dot(A, x), b, atol=eps):
            break
    return x

# ËÆæÁΩÆÁ≥ªÊï∞Áü©Èòµ A ÂíåÂ∏∏Êï∞ÂêëÈáè b
A = np.array([[10, -1, 0, 0], [-1, 10, -1, 0], [0, -1, 10, -1], [0, 0, -1, 10]])
b = np.array([1, 0, 1, 2])

# Ë∞ÉÁî® Gauss-Seidel Ëø≠‰ª£ÂáΩÊï∞Âπ∂ËæìÂá∫ÁªìÊûú
