# PCA降维

# 1、自己算

  1. 对样本数据去中心化。
  2. 计算协方差矩阵。
  3. 对协方差矩阵进行特征值分解，得到特征值以及对应的特征向量。
  4. 根据特征值从大到小排序，选择前 k 个主成分。
  5. 生成新的降维数据，即将原始数据点投影到选取的主成分构成的低维空间中。
d = mean(Q)
% Q -> 数据
W = Q - repmat(d,Q_r,1);
% Q_r -> 数据的行数
% cov() 求协方差矩阵
[V, D] = eig(cov(W))
% V -> 特征向量
% D -> 特征值，从小到大。所以应该从后面开始算
% 若降成一维，则
Q1 = W * V(:,3) % 最后一个最大
% 若降成二维，则
Q2 = W * [V(:,3), V(:,2)]

# 2、命令

[coeff, score, latent] = pca(X);
% coeff -> 特征向量
% score -> 降维结果, 降为 k 维取前 k 列。 score[:, 1:k]
% latent -> 特征值, 从大到小。所以从前面开始取

# K均值

[Idx,Ctrs,SumD,D]=kmeans(X,k)
% X -> N*P的数据矩阵，N为数据个数，P为数据属性维度；
% k -> 表示将X划分为几类，为整数
% Idx -> N*1的向量，存储的是每个点的聚类标号
% Ctrs -> k*P的矩阵，存储的是K个聚类质心位置
% SumD -> k*1的向量，存储的是类间所有点与该类质心点距离之和；
% D -> N*k的矩阵，存储的是每个点与所有质心的距离

% 示例
X = [0 0; 1 0; 0 1; 1 1; 2 1; 1 2; 2 2; 3 2; 6 6; 7 6;
	 8 6; 6 7; 7 7; 8 7; 9 7; 7 8; 8 8; 9 8; 8 9; 9 9]
[Idx,Ctrs,SumD,D]=kmeans(X,2)

% 分类
plot(X(Idx==1,1),X(Idx==1,2),'r.','MarkerSize',14)
hold on
plot(X(Idx==2,1),X(Idx==2,2),'b.','MarkerSize',14)

% 聚类中心
plot(Ctrs(:,1),Ctrs(:,2),'yo','MarkerSize',8,'LineWidth',4) 

ts = [4 4; 5 5]
[Sl, Sl_ind] = pdist2(Ctrs, ts, 'euclidean','Smallest',1)

legend('类别1', '类别2', '聚类中心', '测试点')
