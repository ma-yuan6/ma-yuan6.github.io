class Solution:
    def longestWPI(self, hours: List[int]) -> int:
        maxinterval = 0
        presum = 0
        map = {}
        for i in range(len(hours)):
            presum += 1 if hours[i] > 8 else -1
            if presum > 0:
                maxinterval = i + 1
            else:
                if map.get(presum) == None:
                    map[presum] = i
                
                if map.get(presum - 1) != None:
                    j = map[presum - 1]
                    interval = i - j
                    maxinterval = max(maxinterval, interval)
        return maxinterval
